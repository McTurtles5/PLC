--// DOWNLOADER CODE \\

-- Execute this code to download PLC

local StarterGui = game:GetService("StarterGui")
local function notify(title, text)
  StarterGui:SetCore("SendNotification", {Title = title, Text = text})
end

if writefile then
  local s, e = pcall(function()
    notify("Downloading...", "Fetching the latest PLC script")
    local data = game:HttpGet("https://raw.githubusercontent.com/McTurtles5/PLC/main/PLC.IY")
    notify("Installing...", "Fetched from web and is now installing")
    writefile("PLC.iy", data)
    notify("Downloaded!", "PLC was downloaded, run the command 'addplugin PLC' on Infinite Yield!")
  end)
  if not s then
    notify("Error!", "An unexpected error occured and PLC did not install")
    warn("Failed to download!, "..e)
  end
else
  notify("Can't download!", "Your exploit is incompatable and cannot use PLC (missing writefile)")
end
